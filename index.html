void qsi(int *a, int elements) {
  int  piv, beg[MAX_LEVELS], end[MAX_LEVELS], i=0, L, R, tmp ;
  beg[0]=0; end[0]=elements;
  while (i>=0) {
    L=beg[i]; R=end[i]-1;
    if (L < R) {
      piv=a[L];
      while (L < R) {
        while (a[R]>=piv && L < R) R--; if (L < R) a[L++]=a[R];
        while (a[L]<=piv && L < R) L++; if (L < R) a[R--]=a[L]; }
      a[L]=piv; beg[i+1]=L+1; end[i+1]=end[i]; end[i++]=L;
      if (end[i]-beg[i]>end[i-1]-beg[i-1]) {
        tmp=beg[i]; beg[i]=beg[i-1]; beg[i-1]=tmp;
        tmp=end[i]; end[i]=end[i-1]; end[i-1]=tmp; }}
    else {
      i--; }}}

Quciksort Rekursiv


void qs (int *a, int n) {
	int i, j, p, t;
	if (n < 2) return;
	p = a[n / 2];
	for (i = 0, j = n - 1;; i++, j--) {
		while (a[i] < p) i++;
		while (p < a[j]) j--;
		if (i >= j)	break;
		t = a[i];
		a[i] = a[j];
		a[j] = t;
	}
	qs(a, i);
	qs(a + i, n - i);
}

def sort(xs: Array[Int]): Array[Int] =
	  if (xs.length <= 1) xs
	  else {
		val pivot = xs(xs.length / 2)
		Array.concat(
		  sort(xs filter ( x => pivot > x )),
		  xs filter ( x => pivot == x ),
		  sort(xs filter ( x => pivot < x )))
